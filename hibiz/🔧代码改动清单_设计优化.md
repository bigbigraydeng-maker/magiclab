### 改动 3: 增强排版层次

**文件**: 新建 `src/lib/generation/typography-scale.ts`

**创建**:
```typescript
// src/lib/generation/typography-scale.ts
export const typographyScale = {
  // Hero 标题（最大）
  h1: {
    className: "text-5xl md:text-6xl font-bold tracking-tight leading-tight",
    usage: "Hero section 主标题"
  },

  // 主要章节（Testimonials, Stats, Services）
  h2: {
    className: "text-3xl md:text-4xl font-bold leading-tight",
    usage: "主要 section 的标题"
  },

  // 次要章节（Listings, About）
  h3: {
    className: "text-2xl md:text-3xl font-semibold leading-snug",
    usage: "次要 section 的标题"
  },

  // 卡片标题、列表项标题
  h4: {
    className: "text-lg md:text-xl font-semibold",
    usage: "卡片标题、列表项标题"
  },

  // 正文
  body: {
    className: "text-base md:text-lg leading-relaxed",
    usage: "段落、说明文字"
  },

  // 辅助文字（标签、补充说明）
  small: {
    className: "text-sm text-gray-600 leading-relaxed",
    usage: "标签、补充说明、角色描述"
  },

  // 很小的文字（说明、脚注）
  tiny: {
    className: "text-xs text-gray-500",
    usage: "脚注、细节说明"
  }
};

// 辅助函数：获取排版 class
export function getHeadingClass(level: "h1" | "h2" | "h3" | "h4"): string {
  return typographyScale[level]?.className || typographyScale.h2.className;
}

export function getBodyClass(variant: "body" | "small" | "tiny" = "body"): string {
  return typographyScale[variant]?.className || typographyScale.body.className;
}
```

**使用在组件中**:

```typescript
// 修改 src/components/microsite/modules/Testimonials.tsx
import { getHeadingClass } from "@/lib/generation/typography-scale";

export default function Testimonials({ content }: Props) {
  return (
    <section className="py-20">
      {/* ✅ 改动：使用 h2 排版类 */}
      <h2 className={getHeadingClass("h2")}>
        {content.heading || "客户见证"}
      </h2>

      <div className="mt-12 grid gap-6 sm:grid-cols-2 lg:grid-cols-3">
        {content.items?.map((item) => (
          <div key={item.id} className="rounded-xl border border-gray-200 bg-white p-6">
            {/* 见证内容 */}
            <p className="text-gray-700 italic leading-relaxed">
              &quot;{item.quote}&quot;
            </p>

            {/* ✅ 改动：使用 h4 排版类表示作者名 */}
            <p className={`${getHeadingClass("h4")} mt-4`}>
              {item.author}
            </p>

            {/* ✅ 改动：使用 small 排版类表示角色 */}
            {item.role && (
              <p className={getBodyClass("small")}>
                {item.role}
              </p>
            )}
          </div>
        ))}
      </div>
    </section>
  );
}
```

**改动方案**:
```typescript
import { getHeadingClass, getBodyClass } from "@/lib/generation/typography-scale";

export default function OfferStats({ content, brandColor }: Props) {
  return (
    <section className="bg-gray-50 py-16 md:py-20">
      <div className="container mx-auto px-4">
        {/* ✅ 改动 1: 标题使用 h2 排版类 */}
        <h2 className={getHeadingClass("h2")}>
          {content.heading || "我们的成就"}
        </h2>

        {/* ✅ 改动 2: 改为网格卡片布局 */}
        <ul className="mt-12 grid gap-6 sm:grid-cols-2 lg:grid-cols-4">
          {content.items?.map((item) => (
            <li
              key={item.label}
              className="group rounded-xl border border-gray-200 bg-white p-6 text-center shadow-sm transition hover:shadow-lg hover:border-gray-300"
            >
              {/* ✅ 改动 3: 数字用品牌色突出，使用大字体 */}
              <div
                className="text-4xl md:text-5xl font-bold mb-3 transition group-hover:scale-110"
                style={{ color: brandColor?.accent || "#FFD700" }}
              >
                {item.value}
              </div>

              {/* ✅ 改动 4: 标签文字用 small 排版类 */}
              <p className={`${getBodyClass("small")} text-gray-600`}>
                {item.label}
              </p>
            </li>
          ))}
        </ul>
      </div>
    </section>
  );
}
```
